#This is the list of console commands for the TV and Soundbard

#TV Commands
irsend SEND_ONCE TV KEY_POWER
irsend SEND_ONCE TV KEY_VOLUMEUP
irsend SEND_ONCE TV KEY_VOLUMEDOWN

#SoundBar Commands
irsend SEND_ONCE SOUNDBAR KEY_POWER
irsend SEND_ONCE SOUNDBAR KEY_S
irsend SEND_ONCE SOUNDBAR KEY_VOLUMEUP
irsend SEND_ONCE SOUNDBAR KEY_VOLUMEDOWN
irsend SEND_ONCE SOUNDBAR KEY_MUTE
irsend SEND_ONCE SOUNDBAR KEY_RIGHT
irsend SEND_ONCE SOUNDBAR KEY_LEFT
irsend SEND_ONCE SOUNDBAR KEY_NEXT
irsend SEND_ONCE SOUNDBAR KEY_PLAY
irsend SEND_ONCE SOUNDBAR KEY_PREVIOUS
irsend SEND_ONCE SOUNDBAR KEY_SHUFFLE
irsend SEND_ONCE SOUNDBAR KEY_FN1
irsend SEND_ONCE SOUNDBAR KEY_FN2
irsend SEND_ONCE SOUNDBAR KEY_FN3
irsend SEND_ONCE SOUNDBAR KEY_FN3
